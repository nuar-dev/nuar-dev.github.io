MAIN MODULE: Monthly BudgetingApp

    CREATE an empty list of budget entries
    SET the program status to running

    WHILE the program is running DO

        DISPLAY a welcome message
        DISPLAY all current budget entries with their names and amounts

        DISPLAY a menu of options:
            1. Add Income
            2. Add Fixed Expense
            3. Add Variable Expense
            4. Calculate Total and Exit

        PROMPT the user for a choice

        CASE based on user choice:

            OPTION 1:
                ASK the user to enter the name and amount of income
                IF the amount is valid THEN
                    ADD the income to the list of entries
                ELSE
                    DISPLAY an error message
                END IF

            OPTION 2:
                ASK the user to enter the name and amount of a fixed expense
                IF the amount is valid THEN
                    ADD the fixed expense to the list of entries
                ELSE
                    DISPLAY an error message
                END IF

            OPTION 3:
                ASK the user to enter the name, base amount, and multiplier for a variable expense
                IF the inputs are valid THEN
                    ADD the variable expense to the list of entries
                ELSE
                    DISPLAY an error message
                END IF

            OPTION 4:
                COMPUTE the total balance by summing all incomes and subtracting all expenses
                DISPLAY the total balance
                SET the program status to not running

            OTHERWISE:
                DISPLAY "Invalid selection. Please enter a number between 1 and 4."

        END CASE

    END WHILE

    DISPLAY a closing message
    TERMINATE the program

END MODULE
